import java.sql.*;

public class Main {
    public static void main(String[] args) {
        // Registrar el driver de MySQL
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");

            // Establecer la conexión
            String url = "jdbc:mysql://localhost:3306/project";
            String user = "root"; // Tu usuario de MySQL
            String password = "root"; // Tu contraseña de MySQL

            Connection conn = DriverManager.getConnection(url, user, password);
            System.out.println("Conexión exitosa a la base de datos.");

            // Realizar la consulta, etc.
            // Aquí puedes agregar tu código para trabajar con la base de datos

            conn.close(); // Cerrar la conexión al finalizar

        } catch (ClassNotFoundException e) {
            System.out.println("Error al cargar el driver JDBC.");
            e.printStackTrace();
        } catch (SQLException e) {
            System.out.println("Error de conexión a la base de datos.");
            e.printStackTrace();
        }
    }
}
